链表十二个函数

//定义结构体
struct LNode
{
ElemType data;
LNode next
};
typedef LNode LinkList;

/
链表的十二个函数操作 /

void InitList(LinkList &L) //构造一个空的线性表
{
L = (LinkList)malloc(sizeof(LNode)); //产生头结点，并用L指向它
if(!L) //分配失败
exit(OVERFLOW);
L->next = NULL; //头结点的指针域为空
}

void DestroyList(LinkList &L) //销毁线性表，无头结点，L为NULL
{
LinkList q;
while(L) //L指向结点（非空）
{
q = L->next; //q指向首元结点
free(L); //释放头结点
L = q; // L指向原首元结点，现头结点
}
}

void ClearList(LinkList L) //不改变L,将L置为空表 ，还有头指针与头结点
{
LinkList q = L->next; //q指向第一个结点
L->next = NULL; //头结点指针域为空
DestroyList(q); //销毁q所指向的链表
}

Status ListEmpty(LinkList L) //若L为空表返回true否则返回false
{
if(L->next)
return FALSE;
else
return TRUE;
}

int ListLength(LinkList L) //返回L中元素的个数
{
int i = 0; //计数器初始值为0
LinkList p = L->next; //p指向第一个结点
while(p) //未到表尾
{
++i; //计数器＋1
p = p->next; //p指向下一个结点
}
return i;
}

Status GetElem(LinkList L,int i, Elemtype &e) //L为带头结点的单链表的头指针，当第i个元素存在时， 其值赋给e返回ok，否则返回false
{
int j = 1;
LinkList q = L->next; //q指向第一个结点
while(q && j < i) //当i不存在或者p指向第i个结点时退出循环（j==i）
{
j++; //计数器加1
q = q->next; //q指向下一个结点
}
if(!q || j > i) //第i哥结点不存在
return ERROR；
e = q->data; //第i个元素的值赋给e
return OK;
}

int LocateElem(LinkList L, ElemType e, Status(* compare)(Elemtype, Elemtype)) //返回L中第一个满足关系compare（）元素的位序，若不存在返回值为0
{
LinkList q = L->next; //q指向第一个元素
int i = 0; //计数器值为0
while(q) //未到表尾
{
i++;
if(compare(q->data, e)) //这样的元素存在
return i; //返回位序
else
q = q->next
}
return 0; //这样的元素不存在
}

Status PriorElem(LinkList L, ElemType cur_e, ElemType &pre_e) //若cur_e是L中的元素且不是第一个，用pre_e返回它的前驱返回ok否则操作失败返回error
{
LinkList q, p = L->next;
while(p && p->next) //p所指的节点有后继
{
q = p->next; //q指向p的后继
if(q->data == cure_e)
{
pre_e = p->data;
return OK;
}
p = q; //p的后继不是cur_e，p后移
}
return ERROR;
}

Status NextElem(LinkList L, ElemType cur_e, ElemType &next_e) //若cur_e是L中的元素且不是第一个，用next_e返回它的后继返回ok否则操作失败返回error
{
LinkList q, p = L->next;
while(p && p->next) //p所指的结点有后继
{
if(p->data == cur_e) //找到cur_e这个元素
{
next_e = p->next->data; //返回它的值
return OK;
}
p = p->next; //p指向下一个结点
}
return ERROR;
}

Status ListInsert(LinkList L, int i, Elemtype e) //在第i个位置之前插入e
{
int j = 0;
LinkList s, p = L; //p指向头结点（不是第一个结点）
while(p && j < i - 1) //寻找第i-1个结点
{
j++;
p = p->next; //p指向下一个结点
}
if(!p || j > i - 1)
return ERROR;
s = (LinkList)malloc(sizeof(LNode)); //生成新的结点
s->data = e; //将e赋给新结点
s->next = p->next; //新结点指向原来第i个结点
p->next = s; //第i-1个结点指向新结点
return OK;
}

Status ListDelete(LinkList L, int i, ElemType &e) //不改变L，删除第i个元素，并用e返回其值
{
int j = 0;
LnkList q, p = L;
while(p->next && j < i - 1) //删除操作是吧i-1个结点指向i+1个结点 ,找到i-1个结点或者到了表尾得时候
{
j++;
p = p-next;
}
if(p->next || j > i - 1) //删除位置不合理
return ERROR;
q = p->next; //q指向被删除结点
p->next = q->next; //待删除结点的前驱指向待删除结点的后继
e = q->data; //将待删除结点的值赋给e
free(q);
return OK;
}

void ListTraverse(LinkList L, void(* visit)(ElemType)) //对L中的每一个元素都调用visit函数
{
LinkList p = L->next;
while(p) //p所指向的结点存在
{
visit (p->data);
p = p->next; //p指向下一个结点
}
printf(“\n”);
}
